# Quick Deploy Commands for SignBridge Landing Page
# Copy and paste these commands in order

# ============================================
# STEP 1: Navigate to the landing page folder
# ============================================
cd c:\SignBridge\landing-page

# ============================================
# STEP 2: Initialize Git (if not already done)
# ============================================
git init

# ============================================
# STEP 3: Add the GitHub remote
# ============================================
git remote add origin https://github.com/jesslearns017/sign-test.git

# ============================================
# STEP 4: Add all files to staging
# ============================================
git add .

# ============================================
# STEP 5: Commit with a message
# ============================================
git commit -m "Initial commit: SignBridge landing page with Deaf Flag colors and bilingual support"

# ============================================
# STEP 6: Push to GitHub
# ============================================
# If your branch is 'main':
git push -u origin main

# OR if your branch is 'master':
git push -u origin master
